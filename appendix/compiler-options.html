  shouldDecodeNewlines,
  shouldDecodeNewlinesForHref,
  delimiters,
  comments,

  // baseOptions

  modules: [

      genData

      genData

    model: function(){},

  isReservedTag,
  getTagNamespace,

  warn = (msg, tip) => {
    (tip ? tips : errors).push(msg)
  }
}
